【【【【【数据库数据导出导入操作步骤】】】】】1，docker ps2，在windows docker.desktop对应容器的terminal中执行    或者     cmd  docker exec -it 4e0521049399 bash	   从容器A中打包数据库数据到maxkb.dump:   pg_dump -h localhost -U root -p 5432 -d maxkb -Fc -f maxkb.dump   -U 用户名   -p 端口   -d 数据库表名3，从容器中向外拷贝数据：	从容器中向外拷贝数据：	cmd窗口中执行docker ps找到对应容器ID	docker cp 4e0521049399:/opt/maxkb/app/maxkb.dump maxkb.dump	从4e0521049399中拷贝出来	docker cp maxkb.dump afbda195ca25:/opt/tstarkb/app/	docker exec -it 4e0521049399 bash	docker exec -it afbda195ca25 bash	psql -U root	\l	\c tstarkb	\dt	SELECT * FROM "user";	\q	exit4, 将容器A中的数据恢复到容器B中   pg_restore -h localhost -U root -p 5432 -d tstarkb --clean --create -Fc maxkb.dump   5, 如果容器A和容器B中的数据库表名不一致，比如A中为maxkb, B中为tstarkb   那么A通过pg_dump -h localhost -U root -p 5432 -d maxkb -Fc -f maxkb.dump导出的数据库表名也为maxkb   在恢复到容器B中，数据库表名也会为maxkb，则需要首先删除容器B中的tstarkb， 然后将maxkb改名为tstarkb   操作步骤如下：    1，将所有连接到 tstarkb 的会话全部踢掉：		SELECT pg_terminate_backend(pid)		FROM pg_stat_activity		WHERE datname = 'tstarkb' AND pid <> pg_backend_pid();		DROP DATABASE tstarkb;	2，ALTER DATABASE maxkb RENAME TO tstarkb;         	导出整个数据库（结构 + 数据）	pg_dump -Fc -f all.dump	只导出结构（不含数据）			pg_dump --schema-only -Fc -f schema_only.dump	只导出数据（不含结构）			pg_dump --data-only -Fc -f data_only.dump	只导出某个表（结构 + 数据）		pg_dump -t tablename -Fc -f one_table.dump	只导出某个表的数据				pg_dump -t tablename --data-only -Fc -f one_table_data.dump         docker ps   docker exec -it 4e0521049399 bash   psql -U root   \q   pg_dump -h localhost -U root -p 5432 -d maxkb -Fc -f maxkb.dump   exit   docker cp 4e0521049399:/opt/maxkb/app/maxkb.dump maxkb.dump   docker cp maxkb.dump 5362c7d33ac9:/opt/maxkb/app/   docker exec -it 5362c7d33ac9 bash   pg_restore -h localhost -U root -p 5432 -d maxkb  --clean -Fc maxkb.dump
